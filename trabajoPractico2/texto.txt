import { useMemo, useState } from 'react'
import './App.css'

const CATEGORIAS = ['Comida', 'Transporte', 'Ocio', 'Servicios', 'Salud', 'Otros', 'Sueldo', 'Freelance', 'Inversiones', 'Regalos']

function BalanceTotal({ value, totalGastos, limiteGastos }) {
  const isPositive = value >= 0
  const limiteNumerico = Number(limiteGastos)
  const limiteActivo = !Number.isNaN(limiteNumerico) && limiteNumerico > 0
  const limiteSuperado = limiteActivo && totalGastos > limiteNumerico
  const exceso = limiteSuperado ? totalGastos - limiteNumerico : 0

  return (
    <section className={`balance-card ${limiteSuperado ? 'warning' : ''}`}>
      <p className='balance-label'>Balance Total</p>
      <p className={`balance-amount ${isPositive ? 'positive' : 'negative'}`}>
        ${value.toFixed(2)}
      </p>
      <p className='balance-meta'>Gastos del mes: ${totalGastos.toFixed(2)}</p>
      {limiteSuperado && (
        <p className='warning-text'>
          Limite de gastos superado por ${exceso.toFixed(2)}. Ajusta tus consumos para volver al objetivo mensual.
        </p>
      )}
    </section>
  )
}

function App() {
  const [transacciones, setTransacciones] = useState([])
  const [descripcion, setDescripcion] = useState('')
  const [monto, setMonto] = useState('')
  const [tipo, setTipo] = useState('ingreso')
  const [categoria, setCategoria] = useState(CATEGORIAS[0])
  const [filtroCategoria, setFiltroCategoria] = useState('todas')
  const [limiteGastos, setLimiteGastos] = useState('')
  const [orden, setOrden] = useState('carga_desc')
  const [editandoId, setEditandoId] = useState(null)

  const agregarTransaccion = (e) => {
    e.preventDefault()

    const descripcionLimpia = descripcion.trim()
    const montoNumerico = Number(monto)

    if (!descripcionLimpia || Number.isNaN(montoNumerico) || montoNumerico <= 0) {
      return
    }

    const nueva = {
      id: crypto.randomUUID(),
      descripcion: descripcionLimpia,
      monto: montoNumerico,
      tipo,
      categoria,
      creadoEn: Date.now(),
    }

    if (editandoId) {
      setTransacciones((prev) =>
        prev.map((t) =>
          t.id === editandoId
            ? {
                ...t,
                descripcion: nueva.descripcion,
                monto: nueva.monto,
                tipo: nueva.tipo,
                categoria: nueva.categoria,
              }
            : t,
        ),
      )
      setEditandoId(null)
    } else {
      setTransacciones((prev) => [nueva, ...prev])
    }

    setDescripcion('')
    setMonto('')
    setTipo('ingreso')
    setCategoria(CATEGORIAS[0])
  }

  const iniciarEdicion = (transaccion) => {
    setDescripcion(transaccion.descripcion)
    setMonto(String(transaccion.monto))
    setTipo(transaccion.tipo)
    setCategoria(transaccion.categoria)
    setEditandoId(transaccion.id)
  }

  const cancelarEdicion = () => {
    setDescripcion('')
    setMonto('')
    setTipo('ingreso')
    setCategoria(CATEGORIAS[0])
    setEditandoId(null)
  }

  const eliminarTransaccion = (id) => {
    setTransacciones((prev) => prev.filter((t) => t.id !== id))
    if (editandoId === id) {
      cancelarEdicion()
    }
  }

  const balanceTotal = useMemo(
    () =>
      transacciones.reduce(
        (total, t) => (t.tipo === 'ingreso' ? total + t.monto : total - t.monto),
        0,
      ),
    [transacciones],
  )

  const totalGastos = useMemo(
    () => transacciones.filter((t) => t.tipo === 'gasto').reduce((total, t) => total + t.monto, 0),
    [transacciones],
  )

  const transaccionesFiltradasYOrdenadas = useMemo(() => {
    const filtradas =
      filtroCategoria === 'todas'
        ? [...transacciones]
        : transacciones.filter((t) => t.categoria === filtroCategoria)

    if (orden === 'monto_desc') {
      return filtradas.sort((a, b) => b.monto - a.monto)
    }

    if (orden === 'carga_asc') {
      return filtradas.sort((a, b) => a.creadoEn - b.creadoEn)
    }

    return filtradas.sort((a, b) => b.creadoEn - a.creadoEn)
  }, [filtroCategoria, orden, transacciones])

  return (
    <main className='app'>
      <h1>Gestor de Gastos Personales</h1>
      <p className='subtitle'>Registra ingresos y gastos para seguir tu estado financiero en tiempo real.</p>

      <BalanceTotal value={balanceTotal} totalGastos={totalGastos} limiteGastos={limiteGastos} />

      <section className='panel'>
        <h2>Presupuesto mensual</h2>
        <div className='controls-grid'>
          <div>
            <label htmlFor='limiteGastos'>Limite de gastos</label>
            <input
              id='limiteGastos'
              type='number'
              min='0'
              step='0.01'
              placeholder='Ej: 1500'
              value={limiteGastos}
              onChange={(e) => setLimiteGastos(e.target.value)}
            />
          </div>
        </div>
      </section>

      <section className='panel'>
        <h2>{editandoId ? 'Editar transaccion' : 'Nueva transaccion'}</h2>

        <form className='form' onSubmit={agregarTransaccion}>
          <label htmlFor='descripcion'>Descripcion</label>
          <input
            id='descripcion'
            type='text'
            placeholder='Ej: Sueldo, supermercado, transporte...'
            value={descripcion}
            onChange={(e) => setDescripcion(e.target.value)}
            required
          />

          <label htmlFor='monto'>Monto</label>
          <input
            id='monto'
            type='number'
            min='0.01'
            step='0.01'
            placeholder='0.00'
            value={monto}
            onChange={(e) => setMonto(e.target.value)}
            required
          />

          <label htmlFor='tipo'>Tipo</label>
          <select id='tipo' value={tipo} onChange={(e) => setTipo(e.target.value)}>
            <option value='ingreso'>Ingreso</option>
            <option value='gasto'>Gasto</option>
          </select>

          <label htmlFor='categoria'>Categoria</label>
          <select id='categoria' value={categoria} onChange={(e) => setCategoria(e.target.value)}>
            {CATEGORIAS.map((c) => (
              <option key={c} value={c}>
                {c}
              </option>
            ))}
          </select>

          <button type='submit'>{editandoId ? 'Guardar cambios' : 'Agregar movimiento'}</button>
          {editandoId && (
            <button type='button' className='secondary' onClick={cancelarEdicion}>
              Cancelar edicion
            </button>
          )}
        </form>
      </section>

      <section className='panel'>
        <h2>Movimientos</h2>
        <div className='controls-grid'>
          <div>
            <label htmlFor='filtroCategoria'>Filtrar por categoria</label>
            <select
              id='filtroCategoria'
              value={filtroCategoria}
              onChange={(e) => setFiltroCategoria(e.target.value)}
            >
              <option value='todas'>Todas</option>
              {CATEGORIAS.map((c) => (
                <option key={c} value={c}>
                  {c}
                </option>
              ))}
            </select>
          </div>

          <div>
            <label htmlFor='ordenMovimientos'>Ordenar por</label>
            <select id='ordenMovimientos' value={orden} onChange={(e) => setOrden(e.target.value)}>
              <option value='carga_desc'>Orden de carga (mas recientes)</option>
              <option value='carga_asc'>Orden de carga (mas antiguos)</option>
              <option value='monto_desc'>Monto (mayor a menor)</option>
            </select>
          </div>
        </div>

        {transaccionesFiltradasYOrdenadas.length === 0 ? (
          <p className='empty'>Todavia no registraste movimientos.</p>
        ) : (
          <ul className='transactions'>
            {transaccionesFiltradasYOrdenadas.map((t) => (
              <li key={t.id} className='transaction-item'>
                <div>
                  <p className='transaction-description'>{t.descripcion}</p>
                  <p className='transaction-type'>
                    {t.tipo === 'ingreso' ? 'Ingreso' : 'Gasto'} | {t.categoria}
                  </p>
                </div>
                <div className='transaction-actions'>
                  <p className={t.tipo === 'ingreso' ? 'income' : 'expense'}>
                    {t.tipo === 'ingreso' ? '+' : '-'}${t.monto.toFixed(2)}
                  </p>
                  <div className='actions-row'>
                    <button type='button' className='edit' onClick={() => iniciarEdicion(t)}>
                      Editar
                    </button>
                    <button type='button' className='delete' onClick={() => eliminarTransaccion(t.id)}>
                      Eliminar
                    </button>
                  </div>
                </div>
              </li>
            ))}
          </ul>
        )}
      </section>
    </main>
  )
}

export default App
